    9: //   The STLplus safe_iterator superclasses. This implements the STLplus safe
   10: //   iterator principles. Data structures can then be built using subclasses
   11: //   of safe_iterator for their iterator objects and they will inherit the
   12: //   safe iterator behaviour.
   13: 
   14: //   The data structure must contain a master iterator for each node in the
   15: //   structure. When an iterator is returned to the user, it must be created
   16: //   by the master iterator. When a node is removed from the data structure,
   17: //   its master iterator is destroyed. This sets all iterators pointing to the
   18: //   master iterator to end iterators.
   19: 
   20: ////////////////////////////////////////////////////////////////////////////////
   21: #include "containers_fixes.hpp"
   22: #include "exceptions.hpp"
   23: 
   24: namespace stlplus
   25: {
   26: 
   27:   ////////////////////////////////////////////////////////////////////////////////
   28:   // internals
   29: 
   30:   template<typename O, typename N>
   31:   class safe_iterator_body;
   32: 
   33:   template<typename O, typename N>
   34:   class safe_iterator;
   35: 
   36:   ////////////////////////////////////////////////////////////////////////////////
   37:   // Master Iterator
   38:   // Create one of these in each node in the data structure
   39:   // Generate iterators by obtaining a safe-iterator object from the master iterator
   40:   ////////////////////////////////////////////////////////////////////////////////
   41: 
   42:   template<typename O, typename N>
   43:   class master_iterator
   44:   {
   45:   public:
   46: 
   47:     // construct a valid master iterator connected to the node
   48:     master_iterator(const O* owner, N* node) throw();
   49: 
   50:     // destructor - disconnects all iterators from the node
   51:     ~master_iterator(void) throw();
   52: 
   53:     // dereference
   54:     N* node(void) const throw();
   55:     const O* owner(void) const throw();
   56: 
   57:     // when you move a node from one owner to another, call this on the node's master iterator
   58:     // this effectively moves all other iterators to the node so that they are owned by the new owner too
   59:     void change_owner(const O* owner) throw();
   60: 
   61:     friend class safe_iterator<O,N>;
   62:   private:
   63:     master_iterator(const master_iterator&) throw();
   64:     master_iterator& operator=(const master_iterator&) throw();
   65:     safe_iterator_body<O,N>* m_body;
   66:   };
   67: 
   68:   ////////////////////////////////////////////////////////////////////////////////
   69:   // Safe Iterator
   70:   ////////////////////////////////////////////////////////////////////////////////
   71: 
   72:   template<typename O, typename N>
   73:   class safe_iterator
   74:   {
   75:   public:
   76: 
   77:     // construct a null iterator
   78:     safe_iterator(void) throw();
   79: 
   80:     // construct a valid iterator by aliasing from the owner node's master iterator
   81:     safe_iterator(const master_iterator<O,N>&) throw();
   82: 
   83:     // copy constructor does aliasing
   84:     safe_iterator(const safe_iterator<O,N>&) throw();
   85: 
   86:     // alias an iterator by assignment
   87:     safe_iterator<O,N>& operator=(const safe_iterator<O,N>&) throw();
   88: 
   89:     // destructor
   90:     ~safe_iterator(void) throw();
   91: 
   92:     // reassignment to another node used in increment/decrement operation
   93:     void set(const master_iterator<O,N>&) throw();
   94: 
   95:     // dereference
   96:     N* node(void) const throw();
   97:     const O* owner(void) const throw();
   98: 
   99:     // change to a null iterator - i.e. one that does not belong to any object
  100:     // this does not affect any other iterators pointing to the same node
  101:     void set_null(void) throw();
  102: 
  103:     ////////////////////////////////////////////////////////////////////////////////
  104:     // operations for clients that do not have a master end iterator
  105:     // alternatively, have a master end iterator as part of the container
  106:     // and call constructor(master_end) or set(master_end)
  107: 
  108:     // construct an end iterator
  109:     safe_iterator(const O* owner) throw();
  110: 
  111:     // change to an end iterator - e.g. as a result of incrementing off the end
  112:     void set_end(void) throw();
  113: 
  114:     ////////////////////////////////////////////////////////////////////////////////
  115:     // tests
  116: 
  117:     // comparison
  118:     bool equal(const safe_iterator<O,N>& right) const throw();
  119:     int compare(const safe_iterator<O,N>& right) const throw();
  120: 
  121:     // a null iterator is one that has not been initialised with a value yet
  122:     // i.e. you just declared it but didn't assign to it
  123:     bool null(void) const throw();
  124: 
  125:     // an end iterator is one that points to the end element of the list of nodes
  126:     // in STL conventions this is one past the last valid element and must not be dereferenced
  127:     bool end(void) const throw();
  128: 
  129:     // a valid iterator is one that can be dereferenced
  130:     // i.e. non-null and non-end
  131:     bool valid(void) const throw();
  132: 
  133:     // check the rules for a valid iterator that can be dereferenced
  134:     // optionally also check that the iterator is owned by the owner
  135:     void assert_valid(void) const throw(null_dereference,end_dereference);
  136:     void assert_valid(const O* owner) const throw(wrong_object,null_dereference,end_dereference);
  137:     // assert the rules for a non-null iterator - i.e. valid or end, values that occur in increment operations
  138:     void assert_non_null(void) const throw(null_dereference);
  139:     // assert that this iterator is owned by this container
  140:     void assert_owner(const O* owner) const throw(wrong_object);
  141: 
  142:     ////////////////////////////////////////////////////////////////////////////////
  143: 
  144:     friend class master_iterator<O,N>;
  145:   private:
  146:     safe_iterator_body<O,N>* m_body;
  147:   };
  148: 
  149:   ////////////////////////////////////////////////////////////////////////////////
  150: 
  151: } // end namespace stlplus
  152: 
  153: #include "safe_iterator.tpp"
  154: #endif
