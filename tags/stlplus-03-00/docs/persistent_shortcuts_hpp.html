    9: //   Short-cut functions for dumping and restoring to common targets. These do
   10: //   the whole dump operation in a single function call.
   11: 
   12: //   They take as their second template argument a dump or restore functor which
   13: //   is then called to perform the dump/restore operation.
   14: 
   15: //   They use an installer callback function to install any polymorphic type
   16: //   handlers required prior to performing the dump/restore. If there are no
   17: //   polymorphic types used in the data structure, then the callback can be set
   18: //   to null (i.e. 0).
   19: 
   20: ////////////////////////////////////////////////////////////////////////////////
   21: #include "persistence_fixes.hpp"
   22: #include "persistent_contexts.hpp"
   23: 
   24: ////////////////////////////////////////////////////////////////////////////////
   25: 
   26: namespace stlplus
   27: {
   28: 
   29:   ////////////////////////////////////////////////////////////////////////////////
   30:   // arbitrary IOStream device
   31:   // must be in binary mode
   32: 
   33:   template<typename T, class D>
   34:   void dump_to_device(const T& source, std::ostream& result, D dump_fn, dump_context::installer installer)
   35:     throw(persistent_dump_failed);
   36: 
   37:   template<typename T, class R>
   38:   void restore_from_device(std::istream& source, T& result, R restore_fn, restore_context::installer installer)
   39:     throw(persistent_restore_failed);
   40: 
   41:   ////////////////////////////////////////////////////////////////////////////////
   42:   // string IO device
   43: 
   44:   template<typename T, class D>
   45:   void dump_to_string(const T& source, std::string& result, D dump_fn, dump_context::installer installer)
   46:     throw(persistent_dump_failed);
   47: 
   48:   template<typename T, class R>
   49:   void restore_from_string(const std::string& source, T& result, R restore_fn, restore_context::installer installer)
   50:     throw(persistent_restore_failed);
   51: 
   52:   ////////////////////////////////////////////////////////////////////////////////
   53:   // file IO device
   54: 
   55:   template<typename T, class D>
   56:   void dump_to_file(const T& source, const std::string& filename, D dump_fn, dump_context::installer installer)
   57:     throw(persistent_dump_failed);
   58: 
   59:   template<typename T, class R>
   60:   void restore_from_file(const std::string& filename, T& result, R restore_fn, restore_context::installer installer)
   61:     throw(persistent_restore_failed);
   62: 
   63:   ////////////////////////////////////////////////////////////////////////////////
   64: 
   65: } // end namespace stlplus
   66: 
   67:   ////////////////////////////////////////////////////////////////////////////////
   68: #include "persistent_shortcuts.tpp"
   69: #endif
