   23:     ~matrix(void) throw();
   24: 
   25:     matrix(const matrix&) throw();
   26:     matrix& operator =(const matrix&) throw();
   27: 
   28:     void resize(unsigned rows, unsigned cols, const T& fill = T()) throw();
   29:

   40:

   43:     void fill_row(unsigned row, const T& item = T()) throw(std::out_of_range);
   44:     void fill_leading_diagonal(const T& item = T()) throw();
   45:     void fill_trailing_diagonal(const T& item = T()) throw();
   46:     void make_identity(const T& one, const T& zero = T()) throw();
   47: 
   48:     void transpose(void) throw();
   49: 
   50:   private:
   51:     unsigned m_rows;
   52:     unsigned m_cols;
   53:     T** m_data;
   54:   };
