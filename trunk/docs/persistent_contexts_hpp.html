   26:   // Internals

   28:   class dump_context_body;

   31:   ////////////////////////////////////////////////////////////////////////////////
   32:   // The format version number currently supported
   33:   ////////////////////////////////////////////////////////////////////////////////
   34: 
   35:   extern unsigned char PersistentVersion;
   36: 
   37:   ////////////////////////////////////////////////////////////////////////////////
   38:   // dump_context controls the formatting of a persistent dump
   39:   ////////////////////////////////////////////////////////////////////////////////
   40: 
   41:   class dump_context
   42:   {
   43:     friend class persistent;
   44:   public:
   45:     //////////////////////////////////////////////////////////////////////////////
   46: 
   47:     // device must be in binary mode
   48:     dump_context(std::ostream& device, unsigned char version = PersistentVersion) throw(persistent_dump_failed);
   49:     ~dump_context(void);
   50: 
   51:     // low level output used to dump a byte
   52:     void put(unsigned char data) throw(persistent_dump_failed);
   53: 
   54:     // access the device, for example to check the error status
   55:     const std::ostream& device(void) const;
   56: 
   57:     // recover the version number of the dumped output
   58:     unsigned char version(void) const;
   59: 
   60:     // test whether the current platform uses little-endian or big-endian addressing of bytes
   61:     // this is used in dump/restore of integers and is exported so that other routines can use it
   62:     bool little_endian(void) const;
   63: 
   64:     // Assist functions for Pointers
   65:     // the return pair value is a flag saying whether this is a new pointer and the magic key to dump to file
   66:     std::pair<bool,unsigned> pointer_map(const void* const pointer);
   67:     // the return pair value is a flag saying whether this is a new object and the magic key to dump to file
   68:     std::pair<bool,unsigned> object_map(const void* const pointer);
   69: 
   70:     // Assist functions for Polymorphous classes (i.e. subclasses) using callback approach
   71:     typedef void (*dump_callback)(dump_context&,const void*);
   72:     unsigned register_callback(const std::type_info& info, dump_callback);
   73:     bool is_callback(const std::type_info& info) const;
   74:     typedef std::pair<unsigned,dump_callback> callback_data;
   75:     callback_data lookup_callback(const std::type_info&) const throw(persistent_illegal_type);
   76: 
   77:     // Assist functions for Polymorphous classes (i.e. subclasses) using interface approach
   78:     unsigned register_interface(const std::type_info& info);
   79:     bool is_interface(const std::type_info& info) const;
   80:     unsigned lookup_interface(const std::type_info&) const throw(persistent_illegal_type);
   81: 
   82:     // Register all Polymorphous classes using either approach by calling an installer callback
   83:     typedef void (*installer)(dump_context&);
   84:     void register_all(installer);
   85: 
   86:   private:
   87:     friend class dump_context_body;
   88:     dump_context_body* m_body;
   89: 
   90:     // disallow copying by making assignment and copy constructor private
   91:     dump_context(const dump_context&);
   92:     dump_context& operator=(const dump_context&);
   93:   };
   94: 
   95:   ////////////////////////////////////////////////////////////////////////////////
   96:   // restore_context controls the reading of the persistent data during a restore
   97: 
   98:   class restore_context
   99:   {
  100:     friend class persistent;
  101:   public:
  102:     //////////////////////////////////////////////////////////////////////////////
  103: 
  104:     // device must be in binary mode
  105:     restore_context(std::istream& device) throw(persistent_restore_failed);
  106:     ~restore_context(void);
  107: 
  108:     // low level input used to restore a byte
  109:     int get(void) throw(persistent_restore_failed);
  110: 
  111:     // access the device, for example to check the error status
  112:     const std::istream& device(void) const;
  113: 
  114:     // access the version number of the input being restored
  115:     unsigned char version(void) const;
  116: 
  117:     // test whether the current platform uses little-endian or big-endian addressing of bytes
  118:     // this is used in dump/restore of integers
  119:     bool little_endian(void) const;
  120: 
  121:     // Assist functions for Pointers
  122:     std::pair<bool,void*> pointer_map(unsigned magic);
  123:     void pointer_add(unsigned magic, void* new_pointer);
  124:     std::pair<bool,void*> object_map(unsigned magic);
  125:     void object_add(unsigned magic, void* new_pointer);
  126: 
  127:     // Assist functions for Polymorphous classes using the callback approach
  128:     typedef void* (*create_callback)(void);
  129:     typedef void (*restore_callback)(restore_context&,void*);
  130:     unsigned register_callback(create_callback,restore_callback);
  131:     bool is_callback(unsigned) const;
  132:     typedef std::pair<create_callback, restore_callback> callback_data;
  133:     callback_data lookup_callback(unsigned) const throw(persistent_illegal_type);
  134: 
  135:     // Assist functions for Polymorphous classes using the interface approach
  136:     unsigned register_interface(persistent*);
  137:     bool is_interface(unsigned) const;
  138:     persistent* lookup_interface(unsigned) const throw(persistent_illegal_type);
  139: 
  140:     // Register all Polymorphous classes using either approach by calling an installer callback
  141:     typedef void (*installer)(restore_context&);
  142:     void register_all(installer);
  143: 
  144:   private:
  145:     friend class restore_context_body;
  146:     restore_context_body* m_body;
  147: 
  148:     typedef std::pair<unsigned,persistent*> interface_data;
  149: 
  150:     // disallow copying by making assignment and copy constructor private
  151:     restore_context(const restore_context&);
  152:     restore_context& operator=(const restore_context&);
  153:   };
  154: 
  155:   ////////////////////////////////////////////////////////////////////////////////
