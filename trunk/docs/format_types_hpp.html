   14: // Integer radix display representations:
   15: //   There are three ways in which the radix is represented:
   16: //     - none - the number is just printed as a number (e.g. 12345). Can be confusing for non-decimal radix
   17: //     - C style - for binary, octal and hex, the C-style prefices 0b, 0 and 0x are used
   18: //       note that this is an unsigned representation
   19: //     - Hash style - in the form radix#value - the value may be signed, e.g. 10#-9
   20: 
   21: enum radix_display_t
   22: {
   23:   radix_none,               // just print the number with no radix indicated
   24:   radix_hash_style,         // none for decimal, hash style for all others
   25:   radix_hash_style_all,     // hash style for all radices including decimal
   26:   radix_c_style,            // C style for hex and octal, none for others
   27:   radix_c_style_or_hash     // C style for hex and octal, none for decimal, hash style for others
   28: };
   29: 
   30: ////////////////////////////////////////////////////////////////////////////////
   31: // Floating-point display representations:
   32: // There are three formats for the printout:
   33: //   - fixed - formatted as a fixed-point number, so no mantissa is printed (equivalent to %f in printf)
   34: //   - floating - formatted as a normalised floating-point number (equivalent to %e in printf)
   35: //   - mixed - formatted as fixed-point if appropriate, otherwise the floating format (equivalent to %g in printf)
   36: 
   37: enum real_display_t
   38: {
   39:   display_fixed,    // %f
   40:   display_floating, // %e
   41:   display_mixed     // %g
   42: };
   43: 
   44: ////////////////////////////////////////////////////////////////////////////////
   45: // Alignment:
   46: //   There are three field alignments:
   47: //     - left aligned - the value is to the left of the field which is padded to the right with spaces
   48: //     - right aligned - the value is to the right of the field which is padded to the left with spaces
   49: //     - centred - the value is in the centre of the field and spaces added to both left and right
   50: 
   51: enum alignment_t
   52: {
   53:   align_left,
   54:   align_right,
   55:   align_centre
   56: };
   57:
