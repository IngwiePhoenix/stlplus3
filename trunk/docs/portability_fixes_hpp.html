   67: ////////////////////////////////////////////////////////////////////////////////
   68: // Problems with redefinition of min/max in various different versions of library headers
   69: ////////////////////////////////////////////////////////////////////////////////
   70: 
   71: // The Windows headers define macros called max/min which conflict with the templates std::max and std::min.
   72: // So, to avoid conflicts, MS removed the std::max/min rather than fixing the problem!
   73: // From Visual Studio .NET (SV7, compiler version 13.00) the STL templates have been added correctly.
   74: // For MFC compatibility, only undef min and max in non-MFC programs - some bits of MFC
   75: // use macro min/max in headers. 
   76: 
   77: // I've created extra template function definitions minimum/maximum that avoid all the problems above
   78: 
   79: namespace stlplus
   80: {
   81:   template<typename T> const T& maximum(const T& l, const T& r) {return l > r ? l : r;}
   82:   template<typename T> const T& minimum(const T& l, const T& r) {return l < r ? l : r;}
   83: }

  116: ////////////////////////////////////////////////////////////////////////////////
  117: // Function for establishing endian-ness
  118: ////////////////////////////////////////////////////////////////////////////////
  119: // Different machine architectures store data using different byte orders.
  120: // This is referred to as Big- and Little-Endian Byte Ordering. 
  121: //
  122: // The issue is: where does a pointer to an integer type actually point?
  123: //
  124: // In both conventions, the address points to the left of the word but:
  125: // Big-Endian - The most significant byte is on the left end of a word
  126: // Little-Endian - The least significant byte is on the left end of a word
  127: //
  128: // Bytes are addressed left to right, so in big-endian order byte 0 is the
  129: // msB, whereas in little-endian order byte 0 is the lsB. For example,
  130: // Intel-based machines store data in little-endian byte order so byte 0 is
  131: // the lsB.
  132: //
  133: // This function establishes byte order at run-time
  134: 
  135: namespace stlplus
  136: {
  137:   bool little_endian(void);
  138: }
