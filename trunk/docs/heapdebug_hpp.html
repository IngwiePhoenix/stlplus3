   10: // A subsystem for tracing and reporting heap memory usage using new/delete
   11: 
   12: ////////////////////////////////////////////////////////////////////////////////
   13: 
   14: #ifdef _MSC_VER
   15: #pragma warning(disable: 4290)
   16: #endif
   17: 
   18: #include <new>
   19: 
   20: ////////////////////////////////////////////////////////////////////////////////
   21: // reproduce the standard definitions of new/delete in case of header inconsistencies
   22: 
   23: // standard new and delete - new can throw a bad_alloc if there is no memory left
   24: void* operator new (size_t) throw(std::bad_alloc);
   25: void* operator new[] (size_t) throw(std::bad_alloc);
   26: void operator delete (void*) throw();
   27: void operator delete[] (void*) throw();
   28: 
   29: // nothrow variants will never throw an exception - so new(nothrow) will return null if out of memory
   30: void* operator new (size_t, const std::nothrow_t&) throw();
   31: void* operator new[] (size_t, const std::nothrow_t&) throw();
   32: void operator delete (void* memory, const std::nothrow_t&) throw();
   33: void operator delete[] (void* memory, const std::nothrow_t&) throw();
   34: 
   35: ////////////////////////////////////////////////////////////////////////////////
   36: // exported diagnostic functions
   37: 
   38: namespace stlplus
   39: {
   40:   // report the state of memory - this is automatically called at exit
   41:   void heapdebug_report(void);
   42: 
   43:   // get the error count
   44:   unsigned heapdebug_errors(void);
   45: 
   46:   // get the number of blocks outstanding, i.e. allocated but not deallocated
   47:   unsigned heapdebug_outstanding(void);
   48: }
   49: 
   50: // memory debugging is switched on by the standard NDEBUG macro also used for assertions
   51: #ifndef NDEBUG
   52: 
   53: ////////////////////////////////////////////////////////////////////////////////
   54: // placement new operators which record filename and lineno information for each allocation
   55: // the macros are provided to make these operators easier to use
   56: 
   57: void* operator new (size_t, const char*, int) throw(std::bad_alloc);
   58: void* operator new[] (size_t, const char*, int) throw(std::bad_alloc);
   59: 
   60: #ifndef HEAPDEBUG_NO_PLACEMENT_DELETE
   61: void operator delete (void*, const char*, int) throw();
   62: void operator delete[] (void*, const char*, int) throw();
   63: #endif
   64: 
   65: // Two ways of using this:
   66: //
   67: // 1) #define STLPLUS_HEAPDEBUG_USE_MACRO_NEW before including this header and
   68: // use the NEW macro in your code - this avoids problems with placement
   69: // versions of new but requires changes to your code
   70: //
   71: // 2) (default) use the new operator in your code and let the macros replace
   72: // them with the debugging versions - note that this requires less changes to
   73: // your code but will not work if your code uses placement new
   74: 
   75: #ifdef STLPLUS_HEAPDEBUG_USE_MACRO_NEW
   76: #  define NEW new(__FILE__,__LINE__)
   77: #else
   78: #  define new new(__FILE__,__LINE__)
   79: #endif
   80: 
   81: #else
   82: 
   83: // non-debug version
   84: // if the user has used macros, map them directly onto the keywords
   85: 
   86: #define NEW new
   87: 
   88: #endif
   89: 
   90: #endif
