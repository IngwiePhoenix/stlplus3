   26: #ifdef _MSC_VER
   27: // Microsoft Visual Studio
   28: // shut up the following irritating warnings
   29: //   4275 - VC6, exported class was derived from a class that was not exported
   30: //   4786 - VC6, identifier string exceeded maximum allowable length and was truncated (only affects debugger)
   31: //   4305 - VC6, identifier type was converted to a smaller type
   32: //   4503 - VC6, decorated name was longer than the maximum the compiler allows (only affects debugger)
   33: //   4309 - VC6, type conversion operation caused a constant to exceeded the space allocated for it
   34: //   4290 - VC6, C++ exception specification ignored
   35: //   4800 - VC6, forcing value to bool 'true' or 'false' (performance warning)
   36: //   4675 - VC7.1, "change" in function overload resolution _might_ have altered program
   37: //   4996 - VC8, 'xxxx' was declared deprecated
   38: #pragma warning(disable: 4275 4786 4305 4503 4309 4290 4800 4675 4996)
   39: #endif
   40: 
   41: #ifdef __BORLANDC__
   42: // Borland
   43: // Shut up the following irritating warnings
   44: //   8008 - Condition is always true.
   45: //          Whenever the compiler encounters a constant comparison that (due to
   46: //          the nature of the value being compared) is always true or false, it
   47: //          issues this warning and evaluates the condition at compile time.
   48: //   8060 - Possibly incorrect assignment.
   49: //          This warning is generated when the compiler encounters an assignment
   50: //          operator as the main operator of a conditional expression (part of
   51: //          an if, while, or do-while statement). This is usually a
   52: //          typographical error for the equality operator.
   53: //   8066 - Unreachable code.
   54: //          A break, continue, goto, or return statement was not followed by a
   55: //          label or the end of a loop or function. The compiler checks while,
   56: //          do, and for loops with a constant test condition, and attempts to
   57: //          recognize loops that can't fall through.
   58: #pragma warn -8008
   59: #pragma warn -8060
   60: #pragma warn -8066
   61: #endif
